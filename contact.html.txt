    // Offer card selector
    const offerChoices = document.querySelectorAll('.offer-choice');
    const sessionSelect = document.getElementById('session');

    offerChoices.forEach(choice => {
      choice.addEventListener('click', function() {
        offerChoices.forEach(c => c.style.borderColor = 'transparent');
        this.style.borderColor = 'rgba(201,168,76,0.7)';
        if (sessionSelect) sessionSelect.value = this.dataset.value;
      });
    });

    // Pre-select from URL param
    const params = new URLSearchParams(window.location.search);
    const offerParam = params.get('offer');
    if (offerParam) {
      if (sessionSelect) sessionSelect.value = offerParam;
      offerChoices.forEach(c => {
        c.style.borderColor = 'transparent';
        if (c.dataset.value === offerParam) c.style.borderColor = 'rgba(201,168,76,0.7)';
      });
    }
